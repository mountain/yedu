% 绘制单个格子的函数
% #1: 字符 (如果是空白则不传)
% #2: 列索引 (0-based)
% #3: 行索引 (0-based)
% #4: 模式 (0=范字, 1=描红, 2=空白)
\cs_new_protected:Nn \mingli_draw_cell:nnnn
  {
    \fp_set:Nn \l_tmpa_fp { (#2) * \GridSize }
    \fp_set:Nn \l_tmpb_fp { - (#3) * \GridSize }

    \coordinate (base) at (\fp_use:N \l_tmpa_fp, \fp_use:N \l_tmpb_fp);

    % 1. 绘制外框
    \draw[GridColor,thick] (base) rectangle ++(\GridSize,-\GridSize);

    % 2. 绘制米字格虚线
    \begin{scope}
      \clip (base) rectangle ++(\GridSize,-\GridSize);
      \draw[GuideColor,dashed] (base) -- ++(\GridSize,-\GridSize);
      \draw[GuideColor,dashed] ($(base)+(0,-\GridSize)$) -- ++(\GridSize,\GridSize);
      \draw[GuideColor,dashed] ($(base)+(\GridSize/2,0)$) -- ++(0,-\GridSize);
      \draw[GuideColor,dashed] ($(base)+(0,-\GridSize/2)$) -- ++(\GridSize,0);
    \end{scope}

    % 3. 绘制文字 (根据模式决定颜色和是否绘制)
    % 模式 0: 范字 (ExampleColor)
    % 模式 1: 描红 (TraceColor)
    % 模式 2: 空白 (不绘制)
    \int_case:nnF { #4 }
      {
        {0} {
             \node[anchor=center,text=ExampleColor,inner~sep=0pt]
             at ($(base)+(\GridSize/2,-\GridSize/2)$)
             { \fontsize{0.9cm}{0.9cm}\selectfont #1 };
            }
        {1} {
             \node[anchor=center,text=TraceColor,inner~sep=0pt]
             at ($(base)+(\GridSize/2,-\GridSize/2)$)
             { \fontsize{0.9cm}{0.9cm}\selectfont #1 };
            }
      }
      { } % Default case: do nothing (for mode 2)
  }

% 渲染单页的函数
% #1: 当前页起始字符索引 (1-based)
\cs_new_protected:Nn \mingli_render_page:n
  {
    \begin{center}
    \begin{tikzpicture}[x=1cm,y=1cm]

      % 循环每一行文字 (注意：一行文字 = 页面上的三行格子)
      \int_step_inline:nnn { 0 } { \l_mingli_text_rows_per_page_int - 1 }
        {
          % ##1 是当前页的第几行文字 (0, 1, 2...)

          % 计算这一行文字对应的起始字符索引
          \int_set:Nn \l_tmpa_int { #1 + (##1 * \PageCols) }

          % 检查是否已经超出总字数，如果超出则停止渲染本页剩余部分
          \int_compare:nNnF { \l_tmpa_int } > { \l_mingli_total_chars_int }
          {
            % 内部循环：渲染这一行的每一列
            \int_step_inline:nnn { 0 } { \PageCols - 1 }
              {
                % ####1 是列索引 (0 to PageCols-1)
                \int_set:Nn \l_tmpb_int { \l_tmpa_int + ####1 } % 当前字符的绝对索引

                % 获取字符，如果索引越界则为空
                \int_compare:nNnTF { \l_tmpb_int } > { \l_mingli_total_chars_int }
                  { \tl_clear:N \l_tmpa_tl }
                  { \tl_set:Nx \l_tmpa_tl { \tl_item:Nn \g_mingli_chars_tl { \l_tmpb_int } } }

                % 只有当字符不为空时才绘制范字和描红，但格子始终要画

                % --- Row A: 范字 (Grid Row = ##1 * 3) ---
                \mingli_draw_cell:nnnn { \l_tmpa_tl } { ####1 } { ##1 * 3 } { 0 }

                % --- Row B: 描红 (Grid Row = ##1 * 3 + 1) ---
                \mingli_draw_cell:nnnn { \l_tmpa_tl } { ####1 } { ##1 * 3 + 1 } { 1 }

                % --- Row C: 练习 (Grid Row = ##1 * 3 + 2) ---
                \mingli_draw_cell:nnnn { }            { ####1 } { ##1 * 3 + 2 } { 2 }
              }
          }
        }

    \end{tikzpicture}
    \end{center}
  }

\NewDocumentCommand{\MakeCopybook}{ m }
  {
    \tl_set:Nn \g_mingli_chars_tl { #1 }
    % 清除空格和换行
    \regex_replace_all:nnN { [\s\x{3000}]+ } { } \g_mingli_chars_tl

    \int_set:Nn \l_mingli_total_chars_int { \tl_count:N \g_mingli_chars_tl }

    % 计算：一页能放多少行文字？ (物理行数 / 3)
    \int_set:Nn \l_mingli_text_rows_per_page_int { \int_div_truncate:nn { \PageRows } { 3 } }

    % 计算：一页总共能放多少个字符？
    \int_set:Nn \l_mingli_chars_per_page_int { \l_mingli_text_rows_per_page_int * \PageCols }

    % 循环分页
    \int_step_inline:nnnn { 1 } { \l_mingli_chars_per_page_int } { \l_mingli_total_chars_int }
      {
        \mingli_render_page:n { ##1 }

        \int_compare:nNnT
          { ##1 + \l_mingli_chars_per_page_int - 1 } < { \l_mingli_total_chars_int }
          { \newpage }
      }
  }
